35  /**
36   * Test the issue description reader.
37   * 
38   * @author Jérémie Huchet
39   */
40  public final class IssueDescriptionBuilderTest {
41  
42      /** The event logger. */
43      private static final Logger LOGGER = LoggerFactory.getLogger(IssueDescriptionBuilderTest.class);
44  
45      /** A file containing a stacktrace. */
46      private static final String STACK_SOURCE_FILE = "/files/md5utilstest_fe2d424fe0cb2b241de18c0e5f57e300.txt";
47  
48      /** The stacktrace. */
49      private String stacktrace;
50  
51      /** The MD5 checksum of the stacktrace. */
52      private String md5;
53  
54      /**
55       * @throws IOException
56       */
57      @Before
58      public void setup() throws IOException {
59  
60          final BufferedReader reader = new BufferedReader(new InputStreamReader(this.getClass()
61                  .getResourceAsStream(STACK_SOURCE_FILE)));
62          final StringBuilder stackBuf = new StringBuilder();
63  
64          final char[] cbuf = new char[512];
65          int len;
66          while ((len = reader.read(cbuf)) != -1) {
67              stackBuf.append(cbuf, 0, len);
68          }
69  
70          stacktrace = stackBuf.toString();
71      }
72  
73      /**
74       * Test build an issue description.
75       */
76      @Test
77      public void testGenerateDescription() {
78  
79          final IssueDescriptionBuilder builder = new IssueDescriptionBuilder(stacktrace);
80          final ErrorOccurrence error = new ErrorOccurrence();
81          error.setReportId(UUID.randomUUID().toString());
82          error.setCrashDate(new Date());
83          error.setRunFor(new Date(64784554));
84          error.setAndroidVersion("2.3.3");
85          error.setVersionCode("16");
86          error.setVersionName("0.3.1");
87          error.setDevice("Nexus One / google / passion");
88          builder.addOccurrence(error);
89          final String description = builder.build();
90  
91          if (LOGGER.isTraceEnabled()) {
92              LOGGER.trace(description);
93          }
94  
95          assertNotNull("check the generated description is not null", description);
96      }
97  }
