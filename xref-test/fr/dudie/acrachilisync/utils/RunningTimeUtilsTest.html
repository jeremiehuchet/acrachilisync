32  public class RunningTimeUtilsTest {
33  
34      /** The event logger. */
35      private static final Logger LOGGER = LoggerFactory.getLogger(RunningTimeUtilsTest.class);
36  
37      private static final int S = 1000;
38  
39      private static final int M = 60 * S;
40  
41      private static final int H = 60 * M;
42  
43      private static final int D = 24 * H;
44  
45      public static List<Object[]> data() throws URISyntaxException {
46  
47          final List<Object[]> data = new ArrayList<Object[]>();
48  
49          data.add(new Object[] { H + 10 * S, "1h 10s" });
50          data.add(new Object[] { 10 * S, "0h 10s" });
51          data.add(new Object[] { D + H + M, "1d 1h 1m" });
52  
53          data.add(new Object[] { H + M + S, "0d 1h 1m 1s" });
54          data.add(new Object[] { D + H + M + S, "1d 1h 1m 1s" });
55  
56          for (int d = 0; d < 14; d += 3) {
57              for (int h = 0; h < 24; h += 3) {
58                  for (int m = 0; m < 60; m += 6) {
59                      for (int s = 0; s < 60; s += 5) {
60                          data.add(new Object[] { d * D + h * H + m * M + s * S,
61                                  String.format("%dd %dh %dm %ds", d, h, m, s) });
62                      }
63                  }
64                  LOGGER.info("generated dataset for d={} and h={}", d, h);
65              }
66          }
67  
68          return data;
69      }
70  
71      @Test
72      public void checkParsedTime() throws ParseException {
73  
74          assertEquals("1h 10s", H + 10 * S, RunningTimeUtils.parseRunningTime("1h 10s").getTime());
75          assertEquals("0h 10s", 10 * S, RunningTimeUtils.parseRunningTime("0h 10s").getTime());
76          assertEquals("1d 1h 1m", D + H + M, RunningTimeUtils.parseRunningTime("1d 1h 1m").getTime());
77          assertEquals("0d 1h 1m 1s", H + M + S, RunningTimeUtils.parseRunningTime("0d 1h 1m 1s")
78                  .getTime());
79          assertEquals("1d 1h 1m 1s", D + H + M + S, RunningTimeUtils.parseRunningTime("1d 1h 1m 1s")
80                  .getTime());
81  
82          for (int d = 0; d < 12; d++) {
83              for (int h = 0; h < 24; h++) {
84                  for (int m = 0; m < 60; m++) {
85                      for (int s = 0; s < 60; s++) {
86                          final int expected = d * D + h * H + m * M + s * S;
87                          final String value = String.format("%dd %dh %dm %ds", d, h, m, s);
88                          assertEquals(value, expected, RunningTimeUtils.parseRunningTime(value)
89                                  .getTime());
90                      }
91                  }
92              }
93              LOGGER.info("day {}/11 finished", d);
94          }
95      }
96  
97      @Test
98      public void checkToString() throws ParseException {
99  
100         assertEquals("1d 1h 1m 1s", "1d 1h 1m 1s",
101                 RunningTimeUtils.toString(new Date(D + H + M + S)));
102         assertEquals("1h 1m 1s", "1h 1m 1s", RunningTimeUtils.toString(new Date(H + M + S)));
103         assertEquals("1m 1s", "1m 1s", RunningTimeUtils.toString(new Date(M + S)));
104         assertEquals("1s", "1s", RunningTimeUtils.toString(new Date(S)));
105 
106         assertEquals("8d 2h 54m 41s", "8d 2h 54m 41s",
107                 RunningTimeUtils.toString(new Date(8 * D + 2 * H + 54 * M + 41 * S)));
108 
109         assertEquals("15d 0h 0m 43s", "15d 0h 0m 43s",
110                 RunningTimeUtils.toString(new Date(15 * D + 43 * S)));
111 
112         assertEquals("0s", "0s", RunningTimeUtils.toString(new Date(0)));
113 
114         assertEquals("0d 2h 0s", "2h 0m 0s", RunningTimeUtils.toString(new Date(2 * H)));
115     }
