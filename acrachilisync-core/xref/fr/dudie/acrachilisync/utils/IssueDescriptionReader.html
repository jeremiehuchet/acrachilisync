23   * Helper class to extract informations from an issue description generated with
24   * {@link IssueDescriptionBuilder}.

26   * @author Jérémie Huchet
27   * @see IssueDescriptionBuilder

31      /** The event logger. */
32      private static final Logger LOGGER = LoggerFactory.getLogger(IssueDescriptionReader.class);

34      /** Regexp : <code>|jsglq4354gjdslg4434|dd/MM/yyyy hh:mm:ss|</code>. */
35      private static final String OCCURR_LINE_PATTERN = "\\|[\\w-]+\\|\\d{2}/\\d{2}/\\d{4} \\d{2}:\\d{2}:\\d{2}\\|";

38      private final String stacktrace;

41      private final String stacktraceMD5;

46      /**
47       * Constructor.
48       * <p>
49       * The description must have been formatted with {@link IssueDescriptionBuilder} in order to
50       * make this reader working.
51       * 
52       * @param pIssue
53       *            the issue
54       * @throws IssueParseException
55       *             unable to parse the issue
56       */
57      public IssueDescriptionReader(final Issue pIssue) throws IssueParseException {
58  
59          stacktraceMD5 = getStacktraceMD5(pIssue);
60          if (!StringUtils.isBlank(pIssue.getDescription())) {
61              occurrences.putAll(parseAcraOccurrencesTable(pIssue.getDescription(), stacktraceMD5));
62              stacktrace = parseStacktrace(pIssue.getDescription(), stacktraceMD5);
63          } else {
64              stacktrace = "";
65              if (LOGGER.isDebugEnabled()) {
66                  LOGGER.debug("Issue {} have an empty description", pIssue.getId());
67              }
68          }
69      }

71      /**
72       * Extracts the stacktrace MD5 custom field value from the issue.
73       * 
74       * @param pIssue
75       *            the issue
76       * @return the stacktrace MD5
77       * @throws IssueParseException
78       *             issue doesn't contains a stack trace md5 custom field
79       */
80      private String getStacktraceMD5(final Issue pIssue) throws IssueParseException {
81  
82          final int stackCfId = ConfigurationManager.getInstance().CHILIPROJECT_STACKTRACE_MD5_CF_ID;
83          final Predicate findStacktraceMD5 = new CustomFieldIdPredicate(stackCfId);
84          final CustomField field = (CustomField) CollectionUtils.find(pIssue.getCustomFields(),
85                  findStacktraceMD5);
86          if (null == field) {
87              throw new IssueParseException(String.format(
88                      "Issue %d doesn't contains a custom_field id=%d", pIssue.getId(), stackCfId));
89          }
90          return field.getValue();
91      }

93      /**
94       * Extracts the list of bug occurrences from the description.
95       * 
96       * @param pDescription
97       *            the issue description
98       * @param pStacktraceMD5
99       *            the stacktrace MD5 hash the issue is related to
100      * @return the ACRA bug occurrences listed in the description
101      * @throws IssueParseException
102      *             malformed issue description
103      */
104     private Map<String, Date> parseAcraOccurrencesTable(final String pDescription,
105             final String pStacktraceMD5) throws IssueParseException {
106 
107         final Map<String, Date> occur = new HashMap<String, Date>();
108 
109         // escape braces { and } to use strings in regexp
110         final String header = IssueDescriptionUtils.getOccurrencesTableHeader();
111         final String escHeader = Pattern.quote(header);
112 
113         // regexp to find occurrences tables
114         final Pattern p = Pattern.compile(escHeader + IssueDescriptionUtils.EOL + "(?:"
115                 + OCCURR_LINE_PATTERN + IssueDescriptionUtils.EOL + "+)+", Pattern.DOTALL
116                 | Pattern.CASE_INSENSITIVE);
117         final Matcher m = p.matcher(pDescription);
118 
119         if (m.find()) {
120             // regexp to find occurrences lines
121             final Pattern pLine = Pattern.compile(OCCURR_LINE_PATTERN);
122             final Matcher mLine = pLine.matcher(m.group());
123             while (mLine.find()) {
124                 final StringTokenizer line = new StringTokenizer(mLine.group(), "|");
125                 final String acraReportId = line.nextToken();
126                 final String acraUserCrashDate = line.nextToken();
127                 try {
128                     occur.put(acraReportId, IssueDescriptionUtils.parseDate(acraUserCrashDate));
129                 } catch (final ParseException e) {
130                     throw new IssueParseException("Unable to parse user crash date of ACRA report "
131                             + acraReportId, e);
132                 }
133             }
134         } else {
135             throw new IssueParseException("No crash occurrence table found in the description");
136         }
137 
138         if (m.find()) {
139             throw new IssueParseException("More than 1 occurrence table found in the description");
140         }
141 
142         if (MapUtils.isEmpty(occur)) {
143             throw new IssueParseException("0 user crash occurrence found in the description");
144         }
145 
146         return occur;
147     }

149     /**
150      * Extracts the bug stacktrace from the description.
151      * 
152      * @param pDescription
153      *            the issue description
154      * @param pStacktraceMD5
155      *            the stacktrace MD5 hash the issue is related to
156      * @return the stacktrace
157      * @throws IssueParseException
158      *             malformed issue description
159      */
160     private String parseStacktrace(final String pDescription, final String pStacktraceMD5)
161             throws IssueParseException {
162 
163         String stacktrace = null;
164 
165         // escape braces { and } to use strings in regexp
166         final String start = "<pre class=\"javastacktrace\">";
167         final String qStart = Pattern.quote(start);
168         final String end = "</pre>";
169         final String qEnd = Pattern.quote(end);
170 
171         final Pattern p = Pattern.compile(qStart + "(.*)" + qEnd, Pattern.DOTALL
172                 | Pattern.CASE_INSENSITIVE);
173         final Matcher m = p.matcher(pDescription);
174 
175         if (m.find()) {
176             stacktrace = m.group(1);
177             // if a start tag or an end tag is found in the stacktrace, then there is a problem
178             if (StringUtils.contains(stacktrace, start) || StringUtils.contains(stacktrace, end)) {
179                 throw new IssueParseException("Invalid stacktrace block");
180             }
181         } else {
182             throw new IssueParseException("0 stacktrace block found in the description");
183         }
184 
185         return stacktrace;
186     }
