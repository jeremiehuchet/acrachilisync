39  /**
40   * Helper class to extract informations from an issue description generated with
41   * {@link IssueDescriptionBuilder}.
42   * 
43   * @author Jérémie Huchet
44   * @see IssueDescriptionBuilder
45   */
46  public final class IssueDescriptionReader {
47  
48      /** The event logger. */
49      private static final Logger LOGGER = LoggerFactory.getLogger(IssueDescriptionReader.class);
50  
51      /**
52       * Regexp :
53       * <code>|jsglq4354gjdslg4434|dd/MM/yyyy hh:mm:ss|1d 2h 21m 55s|2.3.3|15|0.3.1|Nexus One / google / passion|</code>
54       * .
55       */
56      public static final String OCCURR_LINE_PATTERN;
57      static {
58          final StringBuilder buf = new StringBuilder();
59          // |jsglq4354gjdslg4434
60          buf.append("\\|").append("[\\w-]+");
61          // |dd/MM/yyyy hh:mm:ss
62          buf.append("\\|").append("\\d{2}/\\d{2}/\\d{4} \\d{2}:\\d{2}:\\d{2}");
63          // |1d 2h 21m 55s OR |2h 21m 55s OR |21m 55s OR | 55s
64          buf.append("\\|").append("(((\\d{1,2}d )?\\d{1,2}h )?\\d{1,2}m )?\\d{1,2}s");
65          // |2.3.3
66          buf.append("\\|").append("[^|]+");
67          // |15
68          buf.append("\\|").append("\\d+");
69          // |0.3.1
70          buf.append("\\|").append("[^|]+");
71          // |Nexus One / google / passion
72          buf.append("\\|").append("[^|]+");
73          // |
74          buf.append("\\|");
75          OCCURR_LINE_PATTERN = buf.toString();
76      }
77  
78      public static final Pattern DESCRIPTION_VERSION_PATTERN = Pattern
79  
80      .compile("\\s*%\\(acrachilisync-description-version\\)description_version_(\\d+)%\\s*",
81              Pattern.CASE_INSENSITIVE);
82  
83      /** The error stacktrace. */
84      private final String stacktrace;
85  
86      /** The error stacktrace MD5 hash. */
87      private final String stacktraceMD5;
88  
89      /** The occurrence time for each ACRA bug report. */
90      private final List<ErrorOccurrence> occurrences = new ArrayList<ErrorOccurrence>();

92      /**
93       * Constructor.
94       * <p>
95       * The description must have been formatted with {@link IssueDescriptionBuilder} in order to
96       * make this reader working.
97       * 
98       * @param pIssue
99       *            the issue
100      * @throws IssueParseException
101      *             unable to parse the issue
102      */
103     public IssueDescriptionReader(final Issue pIssue) throws IssueParseException {
104 
105         checkDescriptionVersion(pIssue);
106 
107         stacktraceMD5 = getStacktraceMD5(pIssue);
108         if (!StringUtils.isBlank(pIssue.getDescription())) {
109             occurrences.addAll(parseAcraOccurrencesTable(pIssue.getDescription(), stacktraceMD5));
110             stacktrace = parseStacktrace(pIssue.getDescription(), stacktraceMD5);
111         } else {
112             stacktrace = "";
113             if (LOGGER.isDebugEnabled()) {
114                 LOGGER.debug("Issue {} have an empty description", pIssue.getId());
115             }
116         }
117     }

119     /**
120      * Checks the description version tag.
121      * 
122      * @param pIssue
123      *            the issue
124      * @throws IssueParseException
125      *             the description version tag doesn't describe the expected version
126      *             {@link IssueDescriptionUtils#DESCRIPTION_VERSION}
127      */
128     private void checkDescriptionVersion(final Issue pIssue) throws IssueParseException {
129 
130         final Matcher m = DESCRIPTION_VERSION_PATTERN.matcher(pIssue.getDescription());
131         final int version;
132         if (m.find()) {
133             version = Integer.parseInt(m.group(1));
134         } else {
135             // default version is 1
136             version = 1;
137         }
138         if (IssueDescriptionUtils.DESCRIPTION_VERSION != version) {
139             throw new IssueParseException(String.format(
140                     "Issue #%s has unsupported description: %d. Expected version %d ",
141                     pIssue.getId(), version, IssueDescriptionUtils.DESCRIPTION_VERSION));
142         }
143     }

145     /**
146      * Extracts the stacktrace MD5 custom field value from the issue.
147      * 
148      * @param pIssue
149      *            the issue
150      * @return the stacktrace MD5
151      * @throws IssueParseException
152      *             issue doesn't contains a stack trace md5 custom field
153      */
154     private String getStacktraceMD5(final Issue pIssue) throws IssueParseException {
155 
156         final int stackCfId = ConfigurationManager.getInstance().CHILIPROJECT_STACKTRACE_MD5_CF_ID;
157         final Predicate findStacktraceMD5 = new CustomFieldIdPredicate(stackCfId);
158         final CustomField field = (CustomField) CollectionUtils.find(pIssue.getCustomFields(),
159                 findStacktraceMD5);
160         if (null == field) {
161             throw new IssueParseException(String.format(
162                     "Issue %d doesn't contains a custom_field id=%d", pIssue.getId(), stackCfId));
163         }
164         return field.getValue();
165     }

167     /**
168      * Extracts the list of bug occurrences from the description.
169      * 
170      * @param pDescription
171      *            the issue description
172      * @param pStacktraceMD5
173      *            the stacktrace MD5 hash the issue is related to
174      * @return the ACRA bug occurrences listed in the description
175      * @throws IssueParseException
176      *             malformed issue description
177      */
178     private List<ErrorOccurrence> parseAcraOccurrencesTable(final String pDescription,
179             final String pStacktraceMD5) throws IssueParseException {
180 
181         final List<ErrorOccurrence> occur = new ArrayList<ErrorOccurrence>();
182 
183         // escape braces { and } to use strings in regexp
184         final String header = IssueDescriptionUtils.getOccurrencesTableHeader();
185         final String escHeader = Pattern.quote(header);
186 
187         // regexp to find occurrences tables
188         final Pattern p = Pattern.compile(escHeader + IssueDescriptionUtils.EOL + "(?:"
189                 + OCCURR_LINE_PATTERN + IssueDescriptionUtils.EOL + "+)+", Pattern.DOTALL
190                 | Pattern.CASE_INSENSITIVE);
191         final Matcher m = p.matcher(pDescription);
192 
193         if (m.find()) {
194             // regexp to find occurrences lines
195             final Pattern pLine = Pattern.compile(OCCURR_LINE_PATTERN);
196             final Matcher mLine = pLine.matcher(m.group());
197             while (mLine.find()) {
198                 try {
199                     final StringTokenizer line = new StringTokenizer(mLine.group(), "|");
200                     final String acraReportId = line.nextToken();
201                     final String acraUserCrashDate = line.nextToken();
202                     final String acraRunFor = line.nextToken();
203                     final String acraAndroidVersion = line.nextToken();
204                     final String acraVersionCode = line.nextToken();
205                     final String acraVersionName = line.nextToken();
206                     final String acraDevice = line.nextToken();
207                     final ErrorOccurrence error = new ErrorOccurrence();
208                     error.setReportId(acraReportId);
209                     try {
210                         error.setCrashDate(IssueDescriptionUtils.parseDate(acraUserCrashDate));
211                         error.setRunFor(RunningTimeUtils.parseRunningTime(acraRunFor));
212                     } catch (final ParseException e) {
213                         throw new IssueParseException(
214                                 "Unable to parse user crash date of ACRA report " + acraReportId, e);
215                     }
216                     error.setAndroidVersion(acraAndroidVersion);
217                     error.setVersionCode(acraVersionCode);
218                     error.setVersionName(acraVersionName);
219                     error.setDevice(acraDevice);
220                     occur.add(error);
221                 } catch (final NoSuchElementException e) {
222                     throw new IssueParseException("Unable to parse ACRA report line: "
223                             + mLine.group(), e);
224                 }
225             }
226         } else {
227             throw new IssueParseException("No crash occurrence table found in the description");
228         }
229 
230         if (m.find()) {
231             throw new IssueParseException("More than 1 occurrence table found in the description");
232         }
233 
234         if (CollectionUtils.isEmpty(occur)) {
235             throw new IssueParseException("0 user crash occurrence found in the description");
236         }
237 
238         return occur;
239     }

241     /**
242      * Extracts the bug stacktrace from the description.
243      * 
244      * @param pDescription
245      *            the issue description
246      * @param pStacktraceMD5
247      *            the stacktrace MD5 hash the issue is related to
248      * @return the stacktrace
249      * @throws IssueParseException
250      *             malformed issue description
251      */
252     private String parseStacktrace(final String pDescription, final String pStacktraceMD5)
253             throws IssueParseException {
254 
255         String stacktrace = null;
256 
257         // escape braces { and } to use strings in regexp
258         final String start = "<pre class=\"javastacktrace\">";
259         final String qStart = Pattern.quote(start);
260         final String end = "</pre>";
261         final String qEnd = Pattern.quote(end);
262 
263         final Pattern p = Pattern.compile(qStart + "(.*)" + qEnd, Pattern.DOTALL
264                 | Pattern.CASE_INSENSITIVE);
265         final Matcher m = p.matcher(pDescription);
266 
267         if (m.find()) {
268             stacktrace = m.group(1);
269             // if a start tag or an end tag is found in the stacktrace, then there is a problem
270             if (StringUtils.contains(stacktrace, start) || StringUtils.contains(stacktrace, end)) {
271                 throw new IssueParseException("Invalid stacktrace block");
272             }
273         } else {
274             throw new IssueParseException("0 stacktrace block found in the description");
275         }
276 
277         return stacktrace;
278     }
