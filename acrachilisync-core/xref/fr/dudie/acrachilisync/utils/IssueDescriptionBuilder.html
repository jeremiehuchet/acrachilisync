18  package fr.dudie.acrachilisync.utils;
19  
20  import java.util.ArrayList;
21  import java.util.Date;
22  import java.util.List;
23  
24  import org.slf4j.Logger;
25  import org.slf4j.LoggerFactory;
26  
27  import fr.dudie.acrachilisync.model.ErrorOccurrence;
28  
29  /**
30   * Helper class to build a formatted issue description to be able to retrieve informations later.
31   * 
32   * @author Jérémie Huchet
33   * @see IssueDescriptionReader
34   */
35  public final class IssueDescriptionBuilder {
36  
37      /** The event logger. */
38      private static final Logger LOGGER = LoggerFactory.getLogger(IssueDescriptionBuilder.class);
39  
40      /** The error stacktrace. */
41      private final String stacktrace;
42  
43      /** The error stacktrace MD5 hash. */
44      private final String stacktraceMD5;
45  
46      /** The occurrence time for each ACRA bug report. */
47      private final List<ErrorOccurrence> occurrences;
48  
49      /**
50       * Default constructor.
51       * 
52       * @param pStacktrace
53       *            the stacktrace
54       */
55      public IssueDescriptionBuilder(final String pStacktrace) {
56  
57          stacktrace = pStacktrace;
58          stacktraceMD5 = MD5Utils.toMD5hash(pStacktrace);
59          occurrences = new ArrayList<ErrorOccurrence>();
60      }
61  
62      /**
63       * Builds the description.
64       * 
65       * @return the issue description
66       * @see IssueDescriptionUtils
67       */
68      public String build() {
69  
70          final StringBuilder description = new StringBuilder();
71  
72          // append occurrences
73          description.append(IssueDescriptionUtils.getOccurrencesTableHeader()).append('\n');
74          for (final ErrorOccurrence error : occurrences) {
75              description.append(IssueDescriptionUtils.getOccurrencesTableLine(error));
76              description.append('\n');
77          }
78  
79          // append stacktrace
80          description.append("\n\n");
81          description.append("*Stacktrace*").append('\n');
82          description.append("<pre class=\"javastacktrace\">");
83          description.append(stacktrace.trim()).append("</pre>");
84  
85          // append version
86          description.append('\n').append(IssueDescriptionUtils.DESCRIPTION_VERSION_TAG);
87  
88          return description.toString();
89      }
90  
91      /**
92       * Sets the list of occurrences for this bug.
93       * 
94       * @param pOccurrences
95       *            a list with the bug occurrence informations
96       * @see #addOccurrence(String, Date)
97       */
98      public void setOccurrences(final List<ErrorOccurrence> pOccurrences) {
99  
100         occurrences.clear();
101         for (final ErrorOccurrence error : pOccurrences) {
102             addOccurrence(error);
103         }
104     }
105 
106     /**
107      * Adds an occurrence for this bug.
108      * 
109      * @param pError
110      *            the ACRA error informations
111      */
112     public void addOccurrence(final ErrorOccurrence pError) {
113 
114         occurrences.add(pError);
115     }
116 }
