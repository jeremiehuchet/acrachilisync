20  import org.apache.commons.collections.CollectionUtils;
21  import org.apache.commons.collections.Predicate;
22  import org.redmine.ta.beans.Issue;
23  import org.slf4j.Logger;
24  import org.slf4j.LoggerFactory;

26  import fr.dudie.acrachilisync.exception.IssueParseException;
27  import fr.dudie.acrachilisync.model.AcraReport;

29  /**
30   * @author Jérémie Huchet
31   */
32  public final class ChiliprojectUtils {
33  
34      /** The event logger. */
35      private static final Logger LOGGER = LoggerFactory.getLogger(ChiliprojectUtils.class);
36  
37      /**
38       * Private constructor to avoid insantiation.
39       */
40      private ChiliprojectUtils() {
41  
42      }
43  
44      /**
45       * Gets wheter or not the given report has already been synchronized with the given issue.
46       * 
47       * @param pReport
48       *            an ACRA report
49       * @param pIssue
50       *            a Chiliproject issue
51       * @return true if the given report has already been synchronized with the given issue
52       * @throws IssueParseException
53       *             the Date of one of the synchronized issue is unreadable
54       */
55      public static boolean isSynchronized(final AcraReport pReport, final Issue pIssue)
56              throws IssueParseException {
57  
58          final IssueDescriptionReader reader = new IssueDescriptionReader(pIssue);
59          return CollectionUtils.exists(reader.getOccurrences(), new ReportPredicate(pReport));
60      }
61  
62      /**
63       * A predicate to find two reports with the same identifier.
64       * 
65       * @see AcraReport#getId()
66       * @author Jérémie Huchet
67       */
68      private static class ReportPredicate implements Predicate {
69  
70          private final AcraReport report;
71  
72          public ReportPredicate(final AcraReport pReport) {
73  
74              this.report = pReport;
75          }
76  
77          @Override
78          public boolean evaluate(final Object object) {
79  
80              if (object instanceof AcraReport) {
81                  return ((AcraReport) object).getId().equals(this.report.getId());
82              } else {
83                  return false;
84              }
85          }
86  
87      }
88  }
