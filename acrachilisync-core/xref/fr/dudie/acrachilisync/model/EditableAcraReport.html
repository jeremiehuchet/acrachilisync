18  package fr.dudie.acrachilisync.model;
19  
20  import java.io.IOException;
21  
22  import com.google.gdata.data.spreadsheet.ListEntry;
23  import com.google.gdata.util.ServiceException;
24  
25  /**
26   * @author Jérémie Huchet
27   */
28  public final class EditableAcraReport extends AcraReport {
29  
30      /** The Google spreadsheet list entry. */
31      private final ListEntry listEntry;
32  
33      /**
34       * Constructor.
35       * 
36       * @param pEntry
37       *            the Google spreadsheet list entry
38       */
39      public EditableAcraReport(final ListEntry pEntry) {
40  
41          super(pEntry.getCustomElements());
42          this.listEntry = pEntry;
43      }
44  
45      /**
46       * Updates the stacktrace MD5 hash on the remote Google spreadsheet.
47       * 
48       * @throws ServiceException
49       * @throws IOException
50       */
51      public void commitStacktraceMD5() throws IOException, ServiceException {
52  
53          listEntry.getCustomElements().setValueLocal(AcraReportHeader.STACK_TRACE_MD5.tagName(),
54                  getStacktraceMD5());
55          listEntry.update();
56      }
57  }
