116     private void upgrade(final int pOldVersion, final int pNewVersion)
117             throws SynchronizationException {
118 
119         final int project = ConfigurationManager.getInstance().CHILIPROJECT_PROJECT_ID;
120         final int tracker = ConfigurationManager.getInstance().CHILIPROJECT_TRACKER_ID;
121 
122         final Map<String, String> selection = new HashMap<String, String>();
123         selection.put("project_id", String.valueOf(project));
124         selection.put("tracker_id", String.valueOf(tracker));
125         selection.put("status_id", "*");
126         try {
127             final List<Issue> issuesToUpgrade = redmine.getIssues(selection);
128             LOGGER.info("Starting upgrade from version {} to {} on {} issues", new Object[] {
129                     pOldVersion, pNewVersion, CollectionUtils.size(issuesToUpgrade) });
130 
131             for (final Issue issue : issuesToUpgrade) {
132                 upgrade(pOldVersion, pNewVersion, issue);
133             }
134         } catch (final Exception e) {
135             throw new SynchronizationException("can't retrive issues from redmine server", e);
136         }
137     }

139     public void upgrade(final int pOldVersion, final int pNewVersion, final Issue pIssue) {
140 
141         if (pOldVersion == 1 && pNewVersion == 2) {
142             LOGGER.info("Issue #{}: upgrade needed from version {} to {}",
143                     new Object[] { pIssue.getId(), pOldVersion, pNewVersion });
144             try {
145                 upgradeFrom1To2(pIssue);
146             } catch (final DescriptionUpgradeException e) {
147                 LOGGER.error("Can't migrate issue #" + pIssue.getId(), e);
148             }
149         } else {
150             LOGGER.info("Issue #{}: current version is {}, no upgrade needed", pIssue.getId(),
151                     pOldVersion);
152         }
153     }
154 
155     private void upgradeFrom1To2(final Issue pIssue) throws DescriptionUpgradeException {
156 
157         try {
158             final IssueDescriptionReaderV1 reader = new IssueDescriptionReaderV1(pIssue);
159             final Map<String, Date> occurrences = reader.getOccurrences();
160             final List<ErrorOccurrence> errors = toErrorOccurrences(occurrences.keySet());
161             final IssueDescriptionBuilder builder = new IssueDescriptionBuilder(
162                     reader.getStacktrace());
163             builder.setOccurrences(errors);
164             pIssue.setDescription(builder.build());
165             if (LOGGER.isTraceEnabled()) {
166                 final StringBuilder msg = new StringBuilder();
167                 msg.append("\n\n================================= Issue #");
168                 msg.append(+pIssue.getId());
169                 msg.append(" =================================\n");
170                 msg.append(pIssue.getDescription());
171                 LOGGER.trace(msg.toString());
172             }
173             LOGGER.info("update Issue #{}", pIssue.getId());
174             redmine.updateIssue(pIssue);
175         } catch (final Exception e) {
176             throw new DescriptionUpgradeException(pIssue, e);
177         }
178     }

180     private List<ErrorOccurrence> toErrorOccurrences(final Set<String> md5stacktraces)
181             throws MalformedURLException, IOException, ServiceException,
182             MalformedSpreadsheetLineException {
183 
184         final List<ErrorOccurrence> errors = new ArrayList<ErrorOccurrence>();
185 
186         for (final String md5stack : md5stacktraces) {
187             // retrieve ACRA report line
188             final String listFeedUrl = String.format(WORKSHEET_URL_FORMAT,
189                     config.getString("google.spreadsheet.document.key"),
190                     config.getString("google.spreadsheet.worksheet.id"), md5stack);
191             LOGGER.debug("retrieving {}", listFeedUrl);
192             final ListFeed listFeed = googleDoc.getFeed(new URL(listFeedUrl), ListFeed.class);
193             final ListEntry entry = listFeed.getEntries().iterator().next();
194 
195             final AcraReport report = new AcraReport(entry.getCustomElements());
196             errors.add(IssueDescriptionUtils.toErrorOccurrence(report));
197         }
198 
199         return errors;
200     }
