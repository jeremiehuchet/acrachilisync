41  /**
42   * Helper class to extract informations from an issue description generated with
43   * IssueDescriptionBuilder.
44   * 
45   * @author Jérémie Huchet
46   */
47  public final class IssueDescriptionReaderV1 {
48  
49      /** The event logger. */
50      private static final Logger LOGGER = LoggerFactory.getLogger(IssueDescriptionReaderV1.class);
51  
52      /** Regexp : <code>|jsglq4354gjdslg4434|dd/MM/yyyy hh:mm:ss|</code>. */
53      private static final String OCCURR_LINE_PATTERN = "\\|[\\w-]+\\|\\d{2}/\\d{2}/\\d{4} \\d{2}:\\d{2}:\\d{2}\\|";
54  
55      /** The error stacktrace. */
56      private final String stacktrace;
57  
58      /** The error stacktrace MD5 hash. */
59      private final String stacktraceMD5;
60  
61      /** The occurrence time for each ACRA bug report. */
62      private final Map<String, Date> occurrences = new HashMap<String, Date>();
63  
64      /**
65       * Constructor.
66       * <p>
67       * The description must have been formatted with IssueDescriptionBuilder in order to make this
68       * reader working.
69       * 
70       * @param pIssue
71       *            the issue
72       * @throws IssueParseException
73       *             unable to parse the issue
74       */
75      public IssueDescriptionReaderV1(final Issue pIssue) throws IssueParseException {
76  
77          stacktraceMD5 = getStacktraceMD5(pIssue);
78          if (!StringUtils.isBlank(pIssue.getDescription())) {
79              occurrences.putAll(parseAcraOccurrencesTable(pIssue.getDescription(), stacktraceMD5));
80              stacktrace = parseStacktrace(pIssue.getDescription(), stacktraceMD5);
81          } else {
82              stacktrace = "";
83              if (LOGGER.isDebugEnabled()) {
84                  LOGGER.debug("Issue {} have an empty description", pIssue.getId());
85              }
86          }
87      }
88  
89      /**
90       * Extracts the stacktrace MD5 custom field value from the issue.
91       * 
92       * @param pIssue
93       *            the issue
94       * @return the stacktrace MD5
95       * @throws IssueParseException
96       *             issue doesn't contains a stack trace md5 custom field
97       */
98      private String getStacktraceMD5(final Issue pIssue) throws IssueParseException {
99  
100         final int stackCfId = ConfigurationManager.getInstance().CHILIPROJECT_STACKTRACE_MD5_CF_ID;
101         final Predicate findStacktraceMD5 = new CustomFieldIdPredicate(stackCfId);
102         final CustomField field = (CustomField) CollectionUtils.find(pIssue.getCustomFields(),
103                 findStacktraceMD5);
104         if (null == field) {
105             throw new IssueParseException(String.format(
106                     "Issue %d doesn't contains a custom_field id=%d", pIssue.getId(), stackCfId));
107         }
108         return field.getValue();
109     }

111     /**
112      * Extracts the list of bug occurrences from the description.
113      * 
114      * @param pDescription
115      *            the issue description
116      * @param pStacktraceMD5
117      *            the stacktrace MD5 hash the issue is related to
118      * @return the ACRA bug occurrences listed in the description
119      * @throws IssueParseException
120      *             malformed issue description
121      */
122     private Map<String, Date> parseAcraOccurrencesTable(final String pDescription,
123             final String pStacktraceMD5) throws IssueParseException {
124 
125         final Map<String, Date> occur = new HashMap<String, Date>();
126 
127         // escape braces { and } to use strings in regexp
128         final String header = IssueDescriptionUtilsV1.getOccurrencesTableHeader();
129         final String escHeader = Pattern.quote(header);
130 
131         // regexp to find occurrences tables
132         final Pattern p = Pattern.compile(escHeader + IssueDescriptionUtilsV1.EOL + "(?:"
133                 + OCCURR_LINE_PATTERN + IssueDescriptionUtilsV1.EOL + "+)+", Pattern.DOTALL
134                 | Pattern.CASE_INSENSITIVE);
135         final Matcher m = p.matcher(pDescription);
136 
137         if (m.find()) {
138             // regexp to find occurrences lines
139             final Pattern pLine = Pattern.compile(OCCURR_LINE_PATTERN);
140             final Matcher mLine = pLine.matcher(m.group());
141             while (mLine.find()) {
142                 final StringTokenizer line = new StringTokenizer(mLine.group(), "|");
143                 final String acraReportId = line.nextToken();
144                 final String acraUserCrashDate = line.nextToken();
145                 try {
146                     occur.put(acraReportId, IssueDescriptionUtilsV1.parseDate(acraUserCrashDate));
147                 } catch (final ParseException e) {
148                     throw new IssueParseException("Unable to parse user crash date of ACRA report "
149                             + acraReportId, e);
150                 }
151             }
152         } else {
153             throw new IssueParseException("No crash occurrence table found in the description");
154         }
155 
156         if (m.find()) {
157             throw new IssueParseException("More than 1 occurrence table found in the description");
158         }
159 
160         if (MapUtils.isEmpty(occur)) {
161             throw new IssueParseException("0 user crash occurrence found in the description");
162         }
163 
164         return occur;
165     }

167     /**
168      * Extracts the bug stacktrace from the description.
169      * 
170      * @param pDescription
171      *            the issue description
172      * @param pStacktraceMD5
173      *            the stacktrace MD5 hash the issue is related to
174      * @return the stacktrace
175      * @throws IssueParseException
176      *             malformed issue description
177      */
178     private String parseStacktrace(final String pDescription, final String pStacktraceMD5)
179             throws IssueParseException {
180 
181         String stacktrace = null;
182 
183         // escape braces { and } to use strings in regexp
184         final String start = "<pre class=\"javastacktrace\">";
185         final String qStart = Pattern.quote(start);
186         final String end = "</pre>";
187         final String qEnd = Pattern.quote(end);
188 
189         final Pattern p = Pattern.compile(qStart + "(.*)" + qEnd, Pattern.DOTALL
190                 | Pattern.CASE_INSENSITIVE);
191         final Matcher m = p.matcher(pDescription);
192 
193         if (m.find()) {
194             stacktrace = m.group(1);
195             // if a start tag or an end tag is found in the stacktrace, then there is a problem
196             if (StringUtils.contains(stacktrace, start) || StringUtils.contains(stacktrace, end)) {
197                 throw new IssueParseException("Invalid stacktrace block");
198             }
199         } else {
200             throw new IssueParseException("0 stacktrace block found in the description");
201         }
202 
203         return stacktrace;
204     }
